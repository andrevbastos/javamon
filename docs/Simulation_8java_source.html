229 BufferedImage grayImage = new BufferedImage(
230 coloredImage.getWidth(),
231 coloredImage.getHeight(),
232 BufferedImage.TYPE_INT_ARGB
233 );
234
235 for (int y = 0; y < coloredImage.getHeight(); y++) {
236 for (int x = 0; x < coloredImage.getWidth(); x++) {
237 int rgb = coloredImage.getRGB(x, y);
238 int alpha = (rgb >> 24) & 0xFF;
239
240 if (alpha == 0) {
241 grayImage.setRGB(x, y, rgb);
242 } else {
243 int r = (rgb >> 16) & 0xFF;
244 int g = (rgb >> 8) & 0xFF;
245 int b = rgb & 0xFF;
246 int grayValue = (int)(0.299 * r + 0.587 * g + 0.114 * b);
247 int grayRGB = (alpha << 24) | (grayValue << 16) | (grayValue << 8) | grayValue;
248 grayImage.setRGB(x, y, grayRGB);
249 }
250 }
