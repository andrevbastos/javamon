301 public boolean checkCondition() {
302 if (this.condition != null) {
303 switch (this.condition) {
304 case BURN -> {
305 this.hp -= this.hpmax / 16;
306 }
307 case FROZEN -> {
308 this.hp -= this.hpmax / 16;
309 }
310 case CONFUSION -> {
311 Random rn = new Random();
312
313 if (conditionCount == 0) {
314 conditionCount = rn.nextInt(3) + 1;
315 } else if (conditionCount == 1) {
316 this.condition = null;

321 if (roll == 3) {
322 this.hp -= this.hpmax / 16;
323 this.conditionCount--;

325 }
326 }

328 Random rn = new Random();
329 int roll = rn.nextInt(4) + 1;
