246 public boolean checkCondition() {
247 if (this.condition != null) {
248 switch (this.condition) {
249 case BURN -> {
250 this.hp -= this.hpmax / 16;
251 }
252 case FROZEN -> {
253 this.hp -= this.hpmax / 16;
254 }
255 case CONFUSION -> {
256 Random rn = new Random();
257
258 if (conditionCount == 0) {
259 conditionCount = rn.nextInt(3) + 1;
260 } else if (conditionCount == 1) {
261 this.condition = null;
262 this.conditionCount = 0;
263 }
264
265 int roll = rn.nextInt(3) + 1;
266 if (roll == 3) {
267 this.hp -= this.hpmax / 16;
268 this.conditionCount--;
269 return false;
270 }
271 }
272 case PARALYZE -> {
273 Random rn = new Random();
274 int roll = rn.nextInt(4) + 1;
275 if (roll == 4) {
276 return false;
277 }
278 }
279 case SLEEP -> {
280 Random rn = new Random();
281 int roll = rn.nextInt(4) + 1;
282 if (roll == 4) {
283 this.condition = null;
284 } else {
285 return false;
286 }
287 }
288 case POISONED -> {
289 this.hp -= (this.hpmax / 16) * conditionCount;
290 }
291 }
292
293 if (this.hp <= 0) {
294 this.hp = 0;
295 return false;
296 }
297 return true;
298 }
299 return true;
